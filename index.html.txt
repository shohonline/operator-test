    <script>
        // Namuna uchun bazada 5 ta element qoldirdim, bunga o'zingizni povodlaringizni to'liq qo'shib yuborasiz
        const rawPovods = [
            { num: "97", uz: "Қасддан одам ўлдириш", ru: "Умышленное убийство" },
            { num: "98", uz: "Кучли руҳий ҳаяжонланиш ҳолатида қасддан одам ўлдириш", ru: "Умышленное убийство в состоянии сильного душевного волнения" },
            { num: "99", uz: "Онанинг ўз чақалоғини қасддан ўлдириши", ru: "Убийство матерью своего новорожденного ребенка" },
            { num: "100", uz: "Зарурий мудофаа чегарасидан четга чиқиб қасддан одам ўлдириш", ru: "Умышленное убийство при превышении пределов необходимой обороны" },
            { num: "103", uz: "Ўзини ўзи ўлдириш даражасига етказиш", ru: "Доведение до самоубийства" }
        ];

        let questionsQueue = [], currentQuestionIndex = 0, userAnswersLog = [], score = 0;
        let userQuestionsLimit = 10, userTimeLimit = 20, selectedLangMode = 'MIX', timerInterval = null, timeLeft = 20;

        function shuffleArray(array) {
            for (let i = array.length - 1; i > 0; i--) {
                const j = Math.floor(Math.random() * (i + 1));
                [array[i], array[j]] = [array[j], array[i]];
            }
            return array;
        }

        function setLangMode(mode, element) {
            selectedLangMode = mode;
            document.querySelectorAll(".lang-btn").forEach(btn => btn.classList.remove("active"));
            element.classList.add("active");
        }

        function startGame() {
            userQuestionsLimit = parseInt(document.getElementById("setup-questions").value) || 10;
            userTimeLimit = parseInt(document.getElementById("setup-timer").value) || 20;
            if (userQuestionsLimit > rawPovods.length) userQuestionsLimit = rawPovods.length;

            document.getElementById("setup-zone").style.display = "none";
            document.getElementById("game-zone").style.display = "block";
            generateQuiz();
        }

        function generateQuiz() {
            let shuffledList = shuffleArray([...rawPovods]);
            if (selectedLangMode === 'UZ') {
                questionsQueue = shuffledList.map(item => ({ num: item.num, text: item.uz, lang: 'UZ', repeatCount: 0 }));
            } else if (selectedLangMode === 'RU') {
                questionsQueue = shuffledList.map(item => ({ num: item.num, text: item.ru, lang: 'RU', repeatCount: 0 }));
            } else {
                const totalToTake = Math.min(userQuestionsLimit, shuffledList.length);
                const uzLimit = Math.round(totalToTake * 0.60);
                questionsQueue = shuffledList.slice(0, totalToTake).map((item, index) => {
                    const isUz = index < uzLimit;
                    return { num: item.num, text: isUz ? item.uz : item.ru, lang: isUz ? 'UZ' : 'RU', repeatCount: 0 };
                });
                questionsQueue = shuffleArray(questionsQueue);
            }
            if (selectedLangMode !== 'MIX') questionsQueue = questionsQueue.slice(0, userQuestionsLimit);
            showQuestion();
        }

        function showQuestion() {
            clearInterval(timerInterval);
            if (currentQuestionIndex >= questionsQueue.length) { finishGame(); return; }
            const currentQ = questionsQueue[currentQuestionIndex];
            
            document.getElementById("q-counter").innerText = `Savol: ${currentQuestionIndex + 1}/${questionsQueue.length}`;
            document.getElementById("score-counter").innerText = `To'g'ri: ${score}`;
            document.getElementById("progress").style.width = ((currentQuestionIndex / questionsQueue.length) * 100) + "%";
            document.getElementById("question").innerHTML = `<span class="badge badge-${currentQ.lang.toLowerCase()}">${currentQ.lang}</span><br><br>${currentQ.text}`;
            
            document.getElementById("answer-input").value = "";
            document.getElementById("answer-input").focus();
            timeLeft = userTimeLimit;
            updateTimerUI();
            timerInterval = setInterval(countdown, 1000);
        }

        function countdown() {
            timeLeft--; updateTimerUI();
            if (timeLeft <= 0) { clearInterval(timerInterval); handleTimeOut(); }
        }

        function updateTimerUI() {
            document.getElementById("timer").innerText = `Vaqt: ${timeLeft}s`;
        }

        function handleTimeOut() {
            const currentQ = questionsQueue[currentQuestionIndex];
            userAnswersLog.push({ text: currentQ.text, lang: currentQ.lang, userAnswer: "Vaqt tugadi ⏱️", correctAnswer: currentQ.num, status: false });
            if (currentQ.repeatCount === 0) {
                currentQ.repeatCount = 1;
                questionsQueue.splice(currentQuestionIndex + 4, 0, currentQ);
            }
            currentQuestionIndex++; showQuestion();
        }

        function checkAnswer() {
            const userAnswer = document.getElementById("answer-input").value.trim();
            if (userAnswer === "") return;
            clearInterval(timerInterval);

            const currentQ = questionsQueue[currentQuestionIndex];
            const isCorrect = userAnswer === currentQ.num;
            userAnswersLog.push({ text: currentQ.text, lang: currentQ.lang, userAnswer: userAnswer, correctAnswer: currentQ.num, status: isCorrect });

            if (isCorrect) score++;
            else if (currentQ.repeatCount === 0) {
                currentQ.repeatCount = 1;
                questionsQueue.splice(currentQuestionIndex + 4, 0, currentQ);
            }
            currentQuestionIndex++; showQuestion();
        }

        function finishGame() {
            document.getElementById("game-zone").style.display = "none";
            document.getElementById("result-zone").style.display = "block";
            document.getElementById("final-score").innerText = `Jami ${questionsQueue.length} ta savoldan ${score} tasiga to'g'ri javob berdingiz.`;
            
            const statsBody = document.getElementById("stats-body");
            statsBody.innerHTML = "";
            userAnswersLog.forEach(item => {
                const row = document.createElement("tr");
                row.className = item.status ? "row-correct" : "row-wrong";
                row.innerHTML = `<td>[${item.lang}] ${item.text}</td><td>${item.userAnswer}</td><td>${item.correctAnswer}</td>`;
                statsBody.appendChild(row);
            });
        }

        document.getElementById("answer-input").addEventListener("keypress", function(e) { if (e.key === "Enter") checkAnswer(); });
        document.getElementById("submit-btn").addEventListener("click", checkAnswer);
    </script>
